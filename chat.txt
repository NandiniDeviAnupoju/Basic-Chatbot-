<body>
  <div id="chatbox">
    <h2>Simple Chatbot</h2>
    <div id="messages"></div>
    <input type="text" id="userInput" placeholder="Ask me something..." />
    <button onclick="sendMessage()">Send</button>
  </div>

  <script>
    const responses = {
      "hello": "Hi there! How can I help you?",
      "what is your name": "I'm a simple chatbot created with JavaScript.",
      "how are you": "I'm just code, but I'm doing fine!",
      "what is html": "HTML stands for HyperText Markup Language. It's used to structure web pages.",
      "what is css": "CSS stands for Cascading Style Sheets. It styles the HTML content.",
      "what is javascript": "JavaScript is a programming language used to add interactivity to websites.",
      "bye": "Goodbye! Have a great day!",
    };

    function sendMessage() {
      const input = document.getElementById("userInput");
      const msg = input.value.trim();
      if (!msg) return;

      showMessage("user", msg);
      input.value = "";

      const reply = getBotResponse(msg.toLowerCase());
      setTimeout(() => showMessage("bot", reply), 500);
    }

    function showMessage(sender, text) {
      const messages = document.getElementById("messages");
      const div = document.createElement("div");
      div.className = sender;
      div.textContent = `${sender === 'user' ? 'You' : 'Bot'}: ${text}`;
      messages.appendChild(div);
      messages.scrollTop = messages.scrollHeight;
    }

    function getBotResponse(input) {
      return responses[input] || "Sorry, I don't understand that. Try asking something else.";
    }
  </script>
